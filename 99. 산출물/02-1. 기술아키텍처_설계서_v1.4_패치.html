정답 확인 처리 흐름

// POST /api/v1/admin/confirm-answer/{pkey} // 관리자가 AI 도출 정답을 확인할 때: async def confirm_answer(pkey: str, admin_user_id: UUID, confirmed: bool, corrected_answer=None): question = await db.get(Question, pkey) if confirmed: # 정답 승인 question.human_confirmed = True question.confirmed_by = admin_user_id question.confirmed_at = now() # 파이프라인: HUMAN_CONFIRM_ANSWER → DATA await orchestrator.transition(pkey, "HUMAN_CONFIRM_ANSWER", "DATA") else: # 정답 수정 필요 → 제작팀으로 반려 if corrected_answer: # 관리자가 직접 올바른 정답을 입력한 경우 produced = await db.get(QuestionProduced, pkey) produced.answer_correct = corrected_answer question.answer_source = "admin_corrected" # 새 소스 타입 question.human_confirmed = True question.confirmed_by = admin_user_id question.confirmed_at = now() await orchestrator.transition(pkey, "HUMAN_CONFIRM_ANSWER", "DATA") else: # 정답 재도출 필요 → PRODUCTION으로 반려 await orchestrator.reject(pkey, "HUMAN_CONFIRM_ANSWER", "PRODUCTION", reject_context={"reason": "관리자가 AI 도출 정답을 부적절하다고 판단"})
answer_source 최종 값 목록: answer_sheet(정답지 추출) / ai_derived(AI 도출, 확인 전) / teacher_input(교사 직접 입력) / admin_corrected(관리자가 정답 수정). 경로 B(ai_derived) 문항은 human_confirmed=true가 되어야 DATA 단계로 진행 가능.
패치 #2

xAPI Config 저장 전략

데이터팀(#7) output인 xapi_config의 저장 위치. xAPI 가이드 수령 전이므로 임시 전략을 명시합니다.

xAPI 저장 전략 (2단계)

Phase 1 (MVP, xAPI 가이드 수령 전)

-- question_produced 테이블에 임시 JSONB 컬럼 추가 ALTER TABLE question_produced ADD COLUMN xapi_config JSONB DEFAULT NULL; -- NULL이면 xAPI 미설정 상태 -- 데이터팀 output을 여기에 임시 저장 -- 데이터검수팀은 이 JSONB의 구조를 검증

Phase 2 (xAPI 가이드 수령 후)

-- 전용 테이블로 분리 CREATE TABLE question_xapi_configs ( id SERIAL PRIMARY KEY, pkey VARCHAR(20) REFERENCES questions(pkey) UNIQUE, events JSONB NOT NULL, -- xAPI 이벤트 정의 배열 interface_mapping JSONB, -- 인터페이스 규격 매핑 test_scenarios JSONB, -- dry-run 시나리오 validated BOOLEAN DEFAULT false, -- 검수 통과 여부 created_at TIMESTAMPTZ ); -- question_produced.xapi_config 데이터를 마이그레이션 후 컬럼 삭제
MVP 시점: xAPI 가이드가 수령되지 않은 상태에서, 데이터팀(#7)과 데이터검수팀(#8) 에이전트는 스킵 처리합니다. 파이프라인: DATA_REVIEW 통과 → EMBEDDING으로 직행. xAPI 가이드 수령 후 해당 에이전트를 활성화하고 기존 문항은 소급 처리.

동기화 구현

// 오케스트레이터가 L2B 상태 전이 시 자동 동기화 async def sync_classroom_exam_status(classroom_exam_id, pipeline_stage): STATUS_MAP = { "DEPLOY_REQUESTED": "PREPARING", "HWP_GENERATING": "PREPARING", "HWP_REVIEW": "PREPARING", "DEPLOY_READY": "READY", "SCHEDULED": "SCHEDULED", "ACTIVE": "ACTIVE", "CLOSED": "CLOSED", "DEPLOY_CANCELLED": "CANCELLED", } new_status = STATUS_MAP[pipeline_stage] await db.execute( "UPDATE classroom_exams SET status = :s WHERE id = :id", {"s": new_status, "id": classroom_exam_id} ) // 이 함수는 오케스트레이터의 handleL2BTransition에서 매 전이마다 호출
SCHEDULED → ACTIVE 자동 전환: 오케스트레이터가 1분 간격으로 classroom_exams WHERE status='SCHEDULED' AND opens_at <= now()를 폴링하여 ACTIVE로 전환. 또는 Redis sorted set에 opens_at을 score로 등록하고 cron-like 체크.
ACTIVE → CLOSED 자동 전환: 마찬가지로 WHERE status='ACTIVE' AND closes_at <= now() 폴링. CLOSED 전환 시 close_classroom_exam() 함수 호출 → 미제출 세션 자동 처리.

async def start_session(classroom_exam_id, student_id): ce = await db.get(ClassroomExam, classroom_exam_id) # 1. 시험 상태 확인 if ce.status != "ACTIVE": raise ExamNotActiveError(ce.status, ce.opens_at) # 2. 기존 제출 확인 existing = await db.query(Submission).filter_by( classroom_exam_id=classroom_exam_id, student_id=student_id ).first() if existing and existing.submitted_at: raise AlreadySubmittedError(existing.submission_id) if existing and not existing.submitted_at: # 3. 진행 중 세션 복구 saved = await redis.hgetall(f"session:{existing.session_id}:answers") return ResumeResponse(existing.session_id, saved, resumed=True) # 4. 새 세션 생성 session_id = generate_session_id() expires_at = now() + timedelta(minutes=ce.exam.time_limit_minutes) # DB: submission 레코드 생성 (submitted_at=null → 진행 중) submission = Submission( submission_id=session_id, exam_id=ce.exam_id, classroom_exam_id=classroom_exam_id, student_id=student_id, started_at=now() ) await db.save(submission) # Redis: 세션 정보 등록 await redis.hset(f"session:{session_id}:info", mapping={ "exam_id": ce.exam_id, "student_id": str(student_id), "classroom_exam_id": str(classroom_exam_id), "expires_at": expires_at.isoformat() }) # Redis: 만료 타이머 ttl = int((expires_at - now()).total_seconds()) await redis.setex(f"exam_expiry:{session_id}", ttl, "1") # Redis: 활성 세션 등록 await redis.sadd(f"exam:{ce.exam_id}:active", session_id) return NewSessionResponse(session_id, exam_data, questions)
패치 #6

오케스트레이터의 데이터 조립 책임

에이전트는 DB에 직접 접근하지 않습니다. 오케스트레이터가 DB에서 데이터를 조회하여 에이전트에 전달하고, 에이전트의 결과를 DB에 저장합니다.

에이전트별 데이터 조립 규칙
에이전트오케스트레이터가 조립하여 전달하는 데이터에이전트 결과를 저장하는 테이블
#1 파싱팀batches 레코드 + S3 파일 경로 + (정답지 경로)questions(마스터) + question_raw
#2 파싱검수팀question_raw 레코드 + 원본 파일 S3 경로pipeline_history (점수/반려사유)
#3 메타팀question_raw 레코드 + meta_schemas(해당 과목)question_structured + question_metadata
#4 메타검수팀question_structured + question_metadata + question_raw(대조용)pipeline_history
#5 제작팀question_structured + question_metadata + question_raw.answer_from_sheetquestion_produced
#6 제작검수팀question_produced + question_structured + question_raw(원본 대조)pipeline_history
#7 데이터팀question_produced 레코드question_produced.xapi_config
#8 데이터검수팀question_produced.xapi_configpipeline_history
임베딩question_structured.stem_text + question_metadataquestion_embeddings
#9 시험지 구성팀teacher_request + question_pool(필터링된 문항 목록)exams + exam_questions
#10 시험지 검수팀exams + exam_questions + teacher_request.conditionspipeline_history
#11 서비스팀classroom_exams 레코드 + exam_questions(전체 문항 데이터 포함)classroom_exams.hwp_file_path, .exam_qr_path
#12 서비스검수팀classroom_exams(HWP 경로) + exam_questions(원본 대조)pipeline_history
#13 채점팀submissions + submission_answers + question_produced(정답) + question_embeddings(추천용)submission_answers(채점결과) + grade_results
#14 채점검수팀grade_results + submission_answers + question_producedpipeline_history
원칙: 에이전트는 DB 커넥션을 갖지 않습니다. 오케스트레이터가 Redis pipeline:tasks Stream에 메시지를 발행할 때, payload에 필요한 데이터를 모두 직렬화하여 포함합니다. 에이전트는 payload만 처리하고 결과를 pipeline:results로 반환합니다. 오케스트레이터가 결과를 받아 DB에 저장합니다.
예외: 대용량 데이터(이미지 파일, HWP 파일)는 payload에 포함하지 않고, S3 경로만 전달합니다. 에이전트가 S3에서 직접 다운로드합니다. 따라서 에이전트는 S3 접근 권한은 필요합니다.
패치 #7

EXAM_CONFIRMED 롤백 규칙

교사가 확정된 시험지의 콘텐츠를 수정하고 싶을 때의 규칙을 명확히 합니다.

롤백 규칙 상세

규칙 1: classroom_exams 연결 여부에 따른 분기

async def handle_exam_rollback_request(exam_id): linked = await db.query(ClassroomExam).filter_by(exam_id=exam_id).count() if linked == 0: # 어떤 학급에도 배포되지 않은 시험지 → 롤백 가능 exam.status = "EXAM_COMPOSE" # 교사가 문항 교체, 배점 수정 등 자유롭게 가능 return {"action": "ROLLBACK_OK"} else: # 1개 이상의 학급에 배포됨 → 롤백 불가 # 이유: 이미 생성된 HWP, QR, 진행 중인 응시에 영향 return { "action": "ROLLBACK_DENIED", "reason": "이미 학급에 배포된 시험지는 수정할 수 없습니다.", "linked_classrooms": linked, "alternatives": [ "새 시험지를 생성하세요.", "기존 배포를 취소하고 새 시험지로 재배포하세요." ] }

규칙 2: 기존 배포 취소 후 새 시험지로 재배포하는 흐름

// 1. 기존 배포 취소 for ce in classroom_exams_of(exam_id): if ce.status in ["ACTIVE"]: # 진행 중인 시험은 취소 불가! (학생 응시 중) raise CannotCancelActiveExam() elif ce.status in ["PREPARING", "READY", "SCHEDULED"]: # 아직 시작 전 → 취소 가능 ce.status = "CANCELLED" pipeline_states[ce.id].current_stage = "DEPLOY_CANCELLED" // 2. 새 시험지 생성 (기존 시험지 복제 + 수정) new_exam = clone_exam(exam_id) # 문항 목록 복사 new_exam.status = "EXAM_COMPOSE" # 수정 가능 상태 # 교사가 수정 후 CONFIRM → 학급에 재배포
ACTIVE 상태 배포는 절대 취소 불가. 학생이 응시 중인 시험을 취소하면 데이터 유실. ACTIVE 배포가 있으면 해당 시험이 CLOSED될 때까지 대기하거나, 정답 정정(ANSWER_CORRECTION)으로 사후 처리.

async def extend_exam_time(ce_id, new_closes_at, extend_sessions): ce = await db.get(ClassroomExam, ce_id) # 1. 기본 검증 if ce.status != "ACTIVE": raise ExamNotActiveError() if new_closes_at <= ce.closes_at: raise ValidationError("연장 시간은 기존 종료 시간보다 이후여야 합니다") # 2. classroom_exams 업데이트 old_closes = ce.closes_at ce.closes_at = new_closes_at await db.save(ce) # 3. 활성 세션 처리 if extend_sessions: active_sessions = await redis.smembers(f"exam:{ce.exam_id}:active") extended_count = 0 for sid in active_sessions: info = await redis.hgetall(f"session:{sid}:info") if not info: continue # 이미 제출/만료된 세션 # 세션별 만료 시간 갱신 old_expires = parse(info["expires_at"]) extension = new_closes_at - old_closes # 연장 시간만큼 추가 new_expires = old_expires + extension await redis.hset(f"session:{sid}:info", "expires_at", new_expires.isoformat()) # exam_expiry TTL 갱신 new_ttl = int((new_expires - now()).total_seconds()) if new_ttl > 0: await redis.setex(f"exam_expiry:{sid}", new_ttl, "1") extended_count += 1 # 4. 이미 자동 제출된 학생 처리 # → 자동 제출된 답안(auto_submitted=true)은 유지됨 # → 재응시 불가 (이미 submitted_at이 있으므로) # → 교사에게 안내: "N명의 학생이 시간 만료로 자동 제출되었습니다" auto_submitted = await db.query(Submission).filter_by( classroom_exam_id=ce_id, auto_submitted=True ).count() return { "extended_sessions": extended_count, "auto_submitted_before_extension": auto_submitted, "message": f"시험 시간이 연장되었습니다. 활성 세션 {extended_count}건 연장. " f"자동 제출 {auto_submitted}건은 복구 불가." }
자동 제출 후 재응시 불가 규칙: 한 번 제출된(auto_submitted=true) 답안은 확정됩니다. 시간 연장 후에도 재응시할 수 없습니다. 이유: ① 이미 제출된 답안과 새 응시 간 점수 비교 문제 ② 다른 학생과의 형평성 ③ 채점 파이프라인에 이미 진입했을 수 있음.
교사 안내 (UI): 시간 연장 버튼 클릭 시 확인 모달: "현재 N명이 응시 중이며, M명이 시간 만료로 자동 제출되었습니다. 자동 제출된 학생의 답안은 복구할 수 없습니다. 시간을 연장하시겠습니까?" 연장은 closes_at **이전에** 하는 것을 권장한다는 안내도 포함.
